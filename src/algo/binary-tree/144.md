function preorderTraversal(root: TreeNode | null): number[] {
  const res: number[] = []

  function traverse(root: TreeNode | null) {
    if (root === null)
      return

    res.push(root.val)
    traverse(root.left)
    traverse(root.right)
  }

  traverse(root)
  return res
}

function preorderTraversal(root: TreeNode | null): number[] {
  const res: number[] = []

  if (root === null)
    return res

  res.push(root.val)
  res.push(...preorderTraversal(root.left))
  res.push(...preorderTraversal(root.right))
  return res
}
