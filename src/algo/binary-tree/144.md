```ts [1]
function preorderTraversal(root: TreeNode | null): number[] {
  let res: number[] = []

  function traverse(root: TreeNode | null) {
    if (root === null)
      return
    
    res.push(root.val)
    traverse(root.left)
    traverse(root.right)
  }

  traverse(root)
  return res
};
```

```ts [2]
function preorderTraversal(root: TreeNode | null): number[] {
  let res: number[] = []

  if (root === null)
    return res

  res.push(root.val)
  res.push(...preorderTraversal(root.left))
  res.push(...preorderTraversal(root.right))
  return res
};
```
